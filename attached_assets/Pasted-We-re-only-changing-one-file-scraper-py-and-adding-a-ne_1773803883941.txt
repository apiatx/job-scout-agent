pythonimport requests
from playwright.sync_api import sync_playwright
from my_criteria import GREENHOUSE_COMPANIES, LEVER_COMPANIES, COMPANY_PAGES

def scrape_greenhouse(slug):
    """Fetch jobs from Greenhouse public API - no login needed"""
    try:
        url = f"https://boards-api.greenhouse.io/v1/boards/{slug}/jobs"
        response = requests.get(url, timeout=10)
        
        if response.status_code != 200:
            print(f"Greenhouse: couldn't find company '{slug}' (status {response.status_code})")
            return None
        
        data = response.json()
        jobs = data.get("jobs", [])
        
        # Convert to plain text so Claude can read it easily
        text_lines = []
        for job in jobs:
            title = job.get("title", "Unknown title")
            location = job.get("location", {}).get("name", "Unknown location")
            job_url = job.get("absolute_url", url)
            text_lines.append(f"JOB: {title} | LOCATION: {location} | URL: {job_url}")
        
        full_text = "\n".join(text_lines)
        print(f"Greenhouse: found {len(jobs)} jobs at {slug}")
        
        return {
            "source_url": f"https://boards-api.greenhouse.io/v1/boards/{slug}/jobs",
            "company_name": slug,
            "raw_text": full_text
        }
        
    except Exception as e:
        print(f"Greenhouse error for {slug}: {e}")
        return None


def scrape_lever(slug):
    """Fetch jobs from Lever public API - no login needed"""
    try:
        url = f"https://api.lever.co/v0/postings/{slug}"
        response = requests.get(url, timeout=10)
        
        if response.status_code != 200:
            print(f"Lever: couldn't find company '{slug}' (status {response.status_code})")
            return None
        
        jobs = response.json()
        
        # Convert to plain text so Claude can read it easily
        text_lines = []
        for job in jobs:
            title = job.get("text", "Unknown title")
            location = job.get("categories", {}).get("location", "Unknown location")
            job_url = job.get("hostedUrl", url)
            team = job.get("categories", {}).get("team", "")
            text_lines.append(f"JOB: {title} | LOCATION: {location} | TEAM: {team} | URL: {job_url}")
        
        full_text = "\n".join(text_lines)
        print(f"Lever: found {len(jobs)} jobs at {slug}")
        
        return {
            "source_url": f"https://api.lever.co/v0/postings/{slug}",
            "company_name": slug,
            "raw_text": full_text
        }
        
    except Exception as e:
        print(f"Lever error for {slug}: {e}")
        return None


def scrape_plain_website(url):
    """Use Playwright to visit a regular careers page as a fallback"""
    try:
        with sync_playwright() as p:
            browser = p.chromium.launch(headless=True)
            page = browser.new_page()
            
            # Pretend to be a real browser so sites don't block us
            page.set_extra_http_headers({
                "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/120.0.0.0 Safari/537.36"
            })
            
            page.goto(url, timeout=30000)
            page.wait_for_timeout(3000)
            text = page.inner_text("body")
            browser.close()
            
            print(f"Playwright: scraped {url}")
            return {
                "source_url": url,
                "company_name": url,
                "raw_text": text[:8000]
            }
            
    except Exception as e:
        print(f"Playwright error for {url}: {e}")
        return None


def scrape_all_jobs():
    """Run all three scrapers and combine results"""
    all_pages = []
    
    # API scrapers first (most reliable)
    print("\n--- Scanning Greenhouse companies ---")
    for slug in GREENHOUSE_COMPANIES:
        result = scrape_greenhouse(slug)
        if result:
            all_pages.append(result)
    
    print("\n--- Scanning Lever companies ---")
    for slug in LEVER_COMPANIES:
        result = scrape_lever(slug)
        if result:
            all_pages.append(result)
    
    # Playwright fallback for plain websites
    if COMPANY_PAGES:
        print("\n--- Scanning plain career pages ---")
        for url in COMPANY_PAGES:
            result = scrape_plain_website(url)
            if result:
                all_pages.append(result)
    
    print(f"\nTotal: scraped {len(all_pages)} company job pages")
    return all_pages
